```ts
function exportCsv(data: ReportData, startDate: string, endDate: string) {
  const rows: string[][] = [
    ["Período", `${startDate} a ${endDate}`],
    [""],
    ["Faturamento"],
    ["Mês", "Receita Bruta", "Custos", "Lucro", "Margem %"],
    ...data.monthly.map((m) => [
      m.month,
      m.revenue.toFixed(2),
      m.costs.toFixed(2),
      m.profit.toFixed(2),
      m.margin.toFixed(1) + "%",
    ]),
    [""],
    ["Faturas"],
    ["Nº", "Cliente", "Valor", "Status", "Vencimento"],
    ...data.invoices.map((inv) => [
      inv.number ?? "",
      inv.clientName ?? "",
      inv.amount.toFixed(2),
      inv.status,
      inv.dueDate ?? "",
    ]),
  ];
  const csv = rows.map((r) => r.map((c) => `"${c}"`).join(",")).join("\n");
  const blob = new Blob(["\uFEFF" + csv], { type: "text/csv;charset=utf-8;" });
  const url = URL.createObjectURL(blob);
  const a = document.createElement("a");
  a.href = url;
  a.download = `relatorio-financeiro-${startDate}-${endDate}.csv`;
  a.click();
  URL.revokeObjectURL(url);
}
```
